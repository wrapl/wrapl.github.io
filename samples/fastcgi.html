  • Using FastCGI

It is possible to write FastCGI applications in Wrapl using the Web.FastCGI module. Typically this involves the following code:

1WITH request <- FastCGI.RequestNew() DO REP ( 2 request:accept; 3 -- Process request here using request:in, request:env and writing to request:out 4 request:finish; 5);

Request is an instance of Web.FastCGI.RequestT.

The function Web.Url.Decode will decode a query string (e.g. request:env("QUERY_STRING") for a GET request or request:in:rest for a POST request) into a table of name/value pairs.

Decoding Form Data

If the form uses the "multipart/form-data" encoding then the data from request:in will have to be decoded. This can be done with the following code (this is example code and may not be complete):

1DEF ProcessRequest(request) ( 2 -- snip 3 VAR type <- request:env("CONTENT_TYPE") | ""; 4 type:begins("multipart/form-data") => ( 5 params <- params + DecodeMultiPart(request:in, type[type:find("boundary=") + 9, 0]); 6 ) // ( 7 params <- params + Url.Decode(request:in:rest); 8 ); 9 -- snip 10); 11 12DEF AttachmentT <- <[] filename, type, tmpfile, size>; 13 14DEF DecodeMultiPart(in, boundary) ( 15 VAR line, name, filename, type, content; 16 VAR i, j; 17 boundary <- "--" + boundary; 18 VAR params <- {}; 19 VAR terminator <- boundary + "--"; 20 REP (WHILE line <- in:read; 21 line:begins("Content-Disposition: form-data") => ( 22 name <- Url.DecodeString(line[line:find('name=\"') + 6, 0]:before('\"')); 23 filename <- Url.DecodeString(line[line:find('filename=\"') + 10, 0]:before('\"')); 24 ) // line:begins("Content-Disposition: file") => ( 25 name <- params:size; 26 filename <- Url.DecodeString(line[line:find('filename=\"') + 10, 0]:before('\"')) | 'file_{name}'; 27 ) // line:begins("Content-Type:") => ( 28 type <- line[15, 0]; 29 ) // line:length = 0 => ( 30 (NIL ~== type):begins("multipart/mixed;") => ( 31 params:insert(name, DecodeMultiPart(in, type[type:find('boundary=') + 9, 0])); 32 line <- in:read; 33 ) // filename ~== NIL => ( 34 i <- 0; 35 content <- FileSys.TempFile(); 36 j <- File.Open(content, File.Flag.Write); 37 REP (WHILE line <- in:readi(0, "\n"); 38 UNTIL line:begins(boundary); 39 i <- i + line:length; 40 j:write(line); 41 ); 42 line[-2, 0] = "\r\n" => line <- line[1, -2]; 43 j:close; 44 params:insert(name, AttachmentT(filename, type, content, i)); 45 ) // ( 46 content <- ""; 47 REP (WHILE line <- in:read; 48 UNTIL line:begins(boundary); 49 content <- $ + Url.DecodeString(line); 50 ); 51 params:insert(name, content); 52 ); 53 name <- type <- filename <- NIL; 54 ); 55 line = terminator => RET params; 56 ); 57 RET params; 58);
